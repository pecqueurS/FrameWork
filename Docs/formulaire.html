Controller.php

<?php

namespace ads\Controllers;

use Bundles\Formulaires\Forms; /* Inclusion de la classe à utiliser */
class Contact {
	
	public function show() {
		$response = array();

		$forgotPwdForm = Forms::make('Contact'); /* Creation du formulaire d'après le json */
		if (!$forgotPwdForm->isValid()) { /* Vérification du formulaire en fonction des contraintes */
			$response['formContact'] = $forgotPwdForm->render(); /* Création du HTML à afficher */
		} else {
			echo "formulaire valide !"; /* Traitement à effectuer */
		}
	
		return $response;
	}
}

Options

Options de base

type

  • textarea (<textarea></textarea>)
  • text (<input type='text'>)
  • email (<input type='email'>)
  • number (<input type='number'>)
  • range (<input type='range'>)
  • money (<input type='number'>)
  • password (<input type='password'>)
  • search (<input type='search'>)
  • url (<input type='url'>)
  • hidden (<input type='hidden'>)
  • date (<input type='date'>)
  • choice (<input type='select|checkbox|radio'>)
  • file (<input type='file'>)
